# =============================================
# GitHub Actions Secrets 설정 가이드
# 
# deploy.yml 파일들이 참조하는 GitHub Secrets 목록입니다.
# GitHub 저장소 → Settings → Secrets and variables → Actions
# → "New repository secret"에서 아래 항목들을 등록하세요.
# =============================================

# -----------------------------------------------
# [공통] Docker Hub 로그인 (3개 서버 모두 사용)
# -----------------------------------------------
# Docker Hub 계정 ID
DOCKER_USERNAME=your_dockerhub_username

# Docker Hub 비밀번호 또는 Access Token
# 발급: https://hub.docker.com/settings/security
DOCKER_PASSWORD=your_dockerhub_password_or_token

# -----------------------------------------------
# [공통] EC2 배포 설정 (3개 서버 모두 사용)
# -----------------------------------------------
# EC2 인스턴스 퍼블릭 IP 또는 도메인 주소
EC2_HOST=your_ec2_public_ip_or_domain

# EC2 SSH 개인키 (전체 내용, -----BEGIN RSA PRIVATE KEY----- 포함)
# EC2 키페어 .pem 파일 내용 전체를 붙여넣기
EC2_SSH_KEY=-----BEGIN RSA PRIVATE KEY-----
            ...your private key content...
            -----END RSA PRIVATE KEY-----

# -----------------------------------------------
# [branding-backend] Spring Boot 환경변수
# EC2에서 docker-compose 실행 시 주입됨
# -----------------------------------------------
DB_URL=jdbc:mysql://your-rds-endpoint:3306/branding
DB_USER=your_db_username
DB_PASSWORD=your_db_password

AWS_ACCESS_KEY=your_aws_iam_access_key
AWS_SECRET_KEY=your_aws_iam_secret_key
AWS_REGION=ap-northeast-2
AWS_S3_BUCKET=your_s3_bucket_name

JWT_SECRET=your_jwt_secret_key_minimum_32_chars

# -----------------------------------------------
# [branding-ai] AI 서버 환경변수
# EC2에서 docker-compose 실행 시 주입됨
# -----------------------------------------------
OPENAI_API_KEY=your_openai_api_key
GEMINI_API_KEY=your_gemini_api_key

CLOUDINARY_CLOUD_NAME=your_cloudinary_cloud_name
CLOUDINARY_API_KEY=your_cloudinary_api_key
CLOUDINARY_API_SECRET=your_cloudinary_api_secret
